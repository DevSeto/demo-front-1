import {Injectable} from "@angular/core";
import {TicketsNotificationModule} from "../models/tickets-notification.module";
import {TicketModule} from "../models/ticket.module";

@Injectable()
export class TicketsNotification  extends Array<TicketsNotificationModule>{



    constructor ()
    {
        super();
        (<any>Object).setPrototypeOf(this, TicketsNotification.prototype);
    }


    public total:number|null = null;
    public total_new:number|null = null;
    public notifications:Array<TicketModule>;


    public set: Function = ( ticketsHistory: TicketsNotificationModule ): void => {
        for ( let history in ticketsHistory )
            this[history] = ticketsHistory[history];
    };

    public reset: Function = (): void => {
        this.splice(0, this.length)
    };
}
