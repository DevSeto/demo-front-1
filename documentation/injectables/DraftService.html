import {Injectable, EventEmitter} from '@angular/core';
import {HttpClient} from '@angular/common/http';
import {GlobalVariables} from '../extra/global.variables';
import {LabelsModel} from "../../models/components/labels.model";
import {toPromise} from "rxjs-compat/operator/toPromise";
import {GlobalServices} from "../helpers/global.service";
import {UsersModel} from "../../models/components/users.model";
import {Label} from '../../dto';
import {LabelModel} from "../../modules/dto/label/models/label.model";
import {DtoMailboxModel} from '../../modules/dto/mailbox/models/mailbox.model';
import {MailBoxService} from './mailbox.service';
import {TicketDtoInterface} from '../../modules/dto/ticket/interfaces/ticket-dto.interface';
import {TicketsService} from './tickets.service';
import {MailBoxModel} from '../../models/components/mailbox.model';
import {SocketService} from './socket.service';
import {UsersService} from './users.service';
import {TicketsModel} from '../../models/components/tickets.model';


@Injectable()
export class DraftService {

    public subdomain: string;
    public user_id: number;
    public socket: any;

    constructor(
        private http: HttpClient,
        private mailBoxModel: MailBoxModel,
        private labelsModel: LabelsModel,
        public ticketService: TicketsService,
        public socketService: SocketService,
        public ticketsModel: TicketsModel,
        public usersService: UsersService,
        public mailboxService: MailBoxService,
        private usersModel: UsersModel,
        public labels: Label
    )
    {
    }

    public draftEvents()
    {
        let self : any = this;

        this.socket.on('deleteIndividualTicketDraft', function( data : any )
        {
            if (
                self.ticketService.ticket
                && self.ticketService.ticket.ticketDraft
                && self.ticketService.ticket.ticketDraft.id == data.data.draft_id
            )
            {
                self.ticketService.ticket.ticketDraft.id = null;
            }

            if(self.ticketService.filterticketsStatus == 'draft' && data.data.draft_id)
            {
                for( let ticketlist in self.ticketService.tickets)
                {
                    if(self.ticketService.tickets[ticketlist]['id'] == data.data.draft_id)
                    {
                        self.ticketService.tickets.splice(ticketlist,1);
                    }
                }

            }

            self.changeInbox(-1, 'draft', data.data.mailbox_id);

        });





        this.socket.on('test', function( data : any ){
            console.log('teeeesy')
        })

        this.socket.on('removeIndividualTicketDraft', function( data : any )
        {
            // console.log(data)
            self.changeInbox(-1,'draft', data.data.mailbox_id);
            if(self.ticketService.filterticketsStatus == 'draft' && data.data.ticketId)
            {
                for( let ticketlist in self.ticketService.tickets)
                {
                    if(self.ticketService.tickets[ticketlist]['id'] == data.data.ticketId)
                    {
                        self.ticketService.tickets.splice(ticketlist,1);
                    }
                }

            }

        });

        this.socket.on('sendIndividualTicketDraftClient', function( data : any )
        {
            let result : any = JSON.parse(data.data);
            let draftData : any = self.usersModel.deCrypt(result.data);
            console.log(draftData);
            let exist : boolean = !!0;
            if(data.status == 'create'){
                self.changeInbox(1,'draft', draftData.mailbox_id);
                if(self.mailBoxModel.getMailboxCurrentId() == draftData.mailbox_id && self.ticketService.filterticketsStatus == 'draft'){
                    if(self.ticketService.tickets.length){
                        self.ticketService.tickets.forEach( ( ticket: any ,index : number) : void =>
                        {
                            if(ticket == draftData){
                                exist = !!1;
                            }
                        })
                    }

                    if(!exist){
                        self.ticketService.tickets.unshift(draftData)
                    }
                }
            }
        });

        this.socket.on('sendCreateTicketDraftClient', function( data : any )
        {
            let result : any = JSON.parse(data.data);
            let draftData : any = self.usersModel.deCrypt(result.data);
            console.log(draftData)
            if(data.status == 'create')
            {
                self.changeInbox(1,'draft', draftData.mailbox_id);
                if(self.mailBoxModel.getMailboxCurrentId() == draftData.mailbox_id && self.ticketService.filterticketsStatus == 'draft')
                {
                    self.ticketService.tickets.unshift(draftData)
                }
            }else if(self.ticketService.filterticketsStatus == 'draft' && self.mailBoxModel.getMailboxCurrentId() == draftData.mailbox_id)
            {
                for( let ticketlist in self.ticketService.tickets)
                {
                    if(!self.ticketService.tickets[ticketlist]['ticket_id'] && self.ticketService.tickets[ticketlist]['id'] == draftData.id)
                    {
                        self.ticketService.tickets[ticketlist] = draftData;
                    }
                }
            }
        });

    }


    /**
     * send data after key up  individual textarea
     */

    public addIndividualdraftData(ticket ? : TicketDtoInterface) : void
    {


        if(!ticket){
            var ticket : TicketDtoInterface = this.ticketService.ticket;
        }else{
            this.socketService.changeTicket(ticket.id,this.user_id,ticket);
        }
        let draftData : any = ticket.ticketDraft;
        draftData.ticket_id = ticket.id;
        draftData.subdomain = this.usersService.company.subdomain;
        draftData.draft_id= draftData.id;
        draftData.forward_emails= draftData.forwarding_emails;
        console.log(draftData)

        this.socket.emit('setindividualdraftData',{
            user_id:  this.user_id,
            ticket:ticket,
            room:this.subdomain,
            user_token :this.usersModel.getToken(),
            encriptData:this.ticketsModel.dataCompile(draftData)

        });
    }

    public closeTicketIndividual(ticketId : number,user_id : number) : void
    {
        this.socket.emit('closeTicketIndividual',{ticketId:ticketId, user_id  : user_id});
    }





    public destroyIndividualPage() : void
    {
        if(this.socket){
            console.log('asdads')
            this.socket.emit('setIndividualDraftData')

        }
    }



    public destroyTicketModal() : void
    {
        console.log(';destroyTicketModal')
        this.socket.emit('sendNewTicketDraftData')
    }


    /**
     * send data after key up all input in form
     * @param {TicketDtoInterface} ticket
     */


    public newTicketDraft( ticket : TicketDtoInterface) : void
    {
        console.log(ticket)
        this.socket.emit('newTicketDraft',{
            user_id:  this.user_id,
            ticket:ticket,
            room:this.subdomain,
            user_token :this.usersModel.getToken(),
            encriptData:this.ticketsModel.dataCompile(ticket)

        });
    }



    public clearTicketDestroy() : void
    {
        this.socket.emit('clearTicketDestroy',{ user_id:  this.user_id})
    }

    public changeInbox(count:number, status :any, mailboxId:number) : void
    {
        this.mailboxService.mailboxes.forEach( ( mailbox: DtoMailboxModel ,index : number) : void =>
        {
            if(mailbox.id == mailboxId && this.mailboxService.mailboxes[index][status+'_tickets_count'] + count >= 0){
                this.mailboxService.mailboxes[index][status+'_tickets_count'] = this.mailboxService.mailboxes[index][status+'_tickets_count'] + count;
            }
        });
    }

}
