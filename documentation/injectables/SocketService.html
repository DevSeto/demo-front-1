import {Injectable, EventEmitter} from '@angular/core';
import {HttpClient} from '@angular/common/http';
import {GlobalVariables} from '../extra/global.variables';
import {LabelsModel} from "../../models/components/labels.model";
import {toPromise} from "rxjs-compat/operator/toPromise";
import {GlobalServices} from "../helpers/global.service";
import {UsersModel} from "../../models/components/users.model";
import {Label} from '../../dto';
import {LabelModel} from "../../modules/dto/label/models/label.model";
import {DtoMailboxModel} from '../../modules/dto/mailbox/models/mailbox.model';
import {MailBoxService} from './mailbox.service';
import {TicketDtoInterface} from '../../modules/dto/ticket/interfaces/ticket-dto.interface';
import {TicketsService} from './tickets.service';
import {MailBoxModel} from '../../models/components/mailbox.model';
import {LabelDtoInterface} from '../../modules/dto/label/interfaces/label-dto.interface';
import {LabelsService} from './labels.service';
import {NotificationJson} from '../../jsons/notification.json'
import {ToastrService} from "../../modules/toastr/toastr/toastr-service";

@Injectable()
export class SocketService {

    public subdomain: string;
    public socket: any;

    public defaultToastrParams: any = <any> {
        progressBar: true,
        positionClass: "toast-bottom-right",
    };

    constructor(
        private http: HttpClient,
        private mailBoxModel: MailBoxModel,
        private labelsModel: LabelsModel,
        public ticketService: TicketsService,
        public labelService: LabelsService,
        public notificationJson: NotificationJson,
        public mailboxService: MailBoxService,
        private userModel: UsersModel,
        public labels: Label,
        private toastr: ToastrService
    )
    {
    }

    public showMsg() {
        this.toastr.info('Ticket has been updated.', 'Info', this.defaultToastrParams);
    }

    public socketEvents()
    {
        let self : any = this;

        this.socket.on('updateStatus', function( data : any )
        {
            if(data)
            {
                var BreakException : any = {};
                self.mailboxService.mailboxes.forEach( ( mailbox: DtoMailboxModel ,index : number) : void => {
                    try {
                        data.mailboxes.forEach( ( newMailbox: DtoMailboxModel ,ind : number) : void => {
                            if (mailbox.id == newMailbox.id) {
                                newMailbox['draft_tickets_count'] =  self.mailboxService.mailboxes[index]['draft_tickets_count'];
                                self.mailboxService.mailboxes[index] = newMailbox;
                                throw BreakException;
                            }
                        })

                    } catch (e ) {
                        if (e !== BreakException) throw e;
                    }
                });

                if(self.ticketService.tickets.length && self.ticketService.filterticketsStatus == data.oldStatus){
                    self.ticketService.tickets.forEach( ( ticket: TicketDtoInterface ,index : number) : void => {
                        try {
                            data.selectedTickets.forEach((selectedTicket: TicketDtoInterface, ind: number): void => {
                                if (ticket.id == selectedTicket.id) {
                                    self.ticketService.tickets.splice(index,1);
                                }
                            })
                        }catch (e ) {
                            if (e !== BreakException) throw e;
                        }
                    })
                }

                data.selectedTickets.forEach((selectedTicket: TicketDtoInterface, ind: number): void =>
                {
                    if(self.mailBoxModel.getMailboxCurrentId() == selectedTicket.mailbox_id){
                        if(self.ticketService.filterticketsStatus == data.currentStatus){
                            selectedTicket.checked = !!0;
                            self.ticketService.tickets.unshift(selectedTicket);

                        }
                        if(self.ticketService.ticket && selectedTicket.id == self.ticketService.ticket.id)
                        {

                            if(self.ticketService.ticket.snooze && data.currentStatus != 'closed'){
                                self.ticketService.ticket.snooze = null;
                            }

                            if( self.ticketService.ticket.status != data.currentStatus ){
                                if(self.ticketService.filterticketsStatus != 'draft'){
                                    self.ticketService.currentTicketsStatus.emit(data.currentStatus);
                                    self.ticketService.filterticketsStatus = data.currentStatus;
                                }

                                selectedTicket.ticketDraft = self.ticketService.ticket.ticketDraft;
                                selectedTicket.status = data.currentStatus;
                                selectedTicket.snooze =  self.ticketService.ticket.snooze;
                                self.ticketService.setIndividual(selectedTicket);
                            }
                            // }

                        }
                    }
                })

                self.showMsg();
            }

        });



        this.socket.on('changeTicket', function( data : any )
        {

            if(self.ticketService.ticket && data.ticket.id == self.ticketService.ticket.id){
                if(data.ticket.status != self.ticketService.ticket.status && self.ticketService.filterticketsStatus != 'draft'){
                    self.ticketService.currentTicketsStatus.emit(data.ticket.status);
                    self.ticketService.filterticketsStatus = data.ticket.status;
                }
                data.ticket.ticketDraft = self.ticketService.ticket.ticketDraft;
                self.ticketService.setIndividual(data.ticket);
            }

            var BreakException : any = {};

            self.mailboxService.mailboxes.forEach( ( mailbox: DtoMailboxModel ,index : number) : void => {
                try {
                    data.mailboxes.forEach( ( newMailbox: DtoMailboxModel ,ind : number) : void => {
                        if (mailbox.id == newMailbox.id) {
                            newMailbox['draft_tickets_count'] =  self.mailboxService.mailboxes[index]['draft_tickets_count'];
                            self.mailboxService.mailboxes[index] = newMailbox;
                            throw BreakException;
                        }
                    })

                } catch (e ) {
                    if (e !== BreakException) throw e;
                }
            });


            if(self.mailBoxModel.getMailboxCurrentId() == data.ticket.mailbox_id && self.ticketService.filterticketsStatus != 'draft'){
                if(self.ticketService.tickets.length ){
                    self.ticketService.tickets.forEach( ( ticket: TicketDtoInterface ,index : number) : void => {
                        if(ticket.id ==  data.ticket.id &&  data.ticket.status != ticket.status){
                            self.ticketService.tickets.splice(index,1);

                        }
                    })

                }
                if(self.ticketService.filterticketsStatus != data.oldStatus && self.ticketService.filterticketsStatus == data.ticket.status){
                    self.ticketService.tickets.unshift(data.ticket);
                }


            }

        });



        this.socket.on('addLabels', function( data : any )
        {
            self.labelService.labels.unshift(data.label);

            self.labelService.labels.forEach((label:LabelDtoInterface,index:number) => {
                if(!label){
                    self.labelService.labels.splice(index,1);
                }
            });

            self.labelService.changedLabelsLength.emit(self.labelService.labels.length);

            self.labelsModel.insert(self.labelService.labels);
        });



        this.socket.on('applyLabelsToTicket', function( data : any )
        {
            console.log(data)
            if(self.ticketService.tickets)
                self.ticketService.tickets.forEach((ticket:any, index:number) => {
                if (data.ticketIds.indexOf(ticket.id) > -1 && !self.hesLabels(self.ticketService.tickets[index],data.label)) {

                    self.ticketService.tickets[index]['labels'].push(data.label);
                    if (ticket.checked){
                        self.labelService.mergeLabel(data.label.id , 1);

                    }
                }
            });


            if(!self.labelService.inactiveLabel.ids[data.label.id] && self.ticketService.ticket && self.ticketService.ticket.id && data.ticketIds.indexOf(self.ticketService.ticket.id)){
                self.labelService.mergeLabel(data.label.id , 1);
                self.ticketService.ticket.labels.push(data.label);

            }

            self.showMsg();
        });

        this.socket.on('removeLabelInTicket', function( data : any )
        {
            self.labelService.mergeLabel(data.labelId , -1);

            if(self.ticketService.ticket && self.ticketService.ticket.id == data.ticketId){
                self.ticketService.ticket.labels.splice(data.index, 1);

            }

            if(self.ticketService.tickets){
                self.ticketService.tickets.forEach((ticket:any, index:number) => {
                    if(ticket.id == data.ticketId){
                        self.ticketService.tickets[index].labels.splice(data.index, 1);
                    }

                })

            }

            self.showMsg();
        });

        this.socket.on('setSnooze', function( data : any )
        {
            if(data){
                data.selectedTickets.forEach((selectedTicket: TicketDtoInterface, ind: number): void =>
                {
                    if(self.ticketService.ticket && selectedTicket.id == self.ticketService.ticket.id)
                    {

                        self.ticketService.ticket.snooze = data.snoozeData;
                            // }
                        self.showMsg();
                    }
                })

            }
        });



        this.socket.on('newNote', function( data : any )
        {

            if(self.ticketService.ticket && data.data.ticket_id == self.ticketService.ticket.id){
                data.data.from_email = data.data.customer_email;
                data.data.from_name = data.data.customer_name;
                console.log( 'newNote',data.data);

                data.data.type = 'note';
                data.data.note = data.data.body;
                data.data.updated_at = data.data.created_at;
                self.ticketService.ticketTimeline.unshift(data.data);
            }
            self.setNotificationInList(data.data);
            // console.log('new note',self.ticketService.ticketsNotification)

        });

        this.socket.on('newReplies', function( data : any )
        {
            console.log('newReplies',data);

            if(self.ticketService.ticket && data.data.ticket_id == self.ticketService.ticket.id){
                data.data.from_email = data.data.customer_email;
                data.data.from_name = data.data.customer_name;

                data.data.type = 'comment';
                self.ticketService.ticketTimeline.unshift(data.data);
            }
            self.setNotificationInList(data.data);
        });

        this.socket.on('newTicket', function( data : any )
        {
            console.log('newTicket',data)
            //backendic piti ga
            data.data.viewed = 0;
            data.data.id = data.data.ticket_id;


            if(!data.data.status){
                data.data.status = 'open';
            }

            if(
                self.mailBoxModel.getMailboxCurrentId() == data.data.mailbox_id && data.data.status == self.ticketService.filterticketsStatus
            )
            {
                self.ticketService.tickets.unshift(data.data)

            }

            self.changeInbox(1,data.data.status, data.data.mailbox_id);
            self.setNotificationInList(data.data);

        });


    }


    private setNotificationInList(data:any) :void
    {
        console.log(data)
        if(data.userData){
            let notificationData :any = {};

            notificationData.notification_id = data.userData.user_notification_id;
            notificationData.is_viewed = 0;
            notificationData.notifications = data;
            console.log(!this.ticketService.ticketsNotification.total)
            if(!this.ticketService.ticketsNotification.total){
                this.ticketService.ticketsNotification.total = 1;
                this.ticketService.ticketsNotification.total_new = 1;
                this.ticketService.ticketsNotification.set({notifications :[notificationData]});
            }else{
                this.ticketService.ticketsNotification.total_new++;
                this.ticketService.ticketsNotification.notifications.unshift(notificationData);
            }
            console.log(this.ticketService.ticketsNotification)
        }

    }

    private hesLabels( ticket : TicketDtoInterface ,label : LabelDtoInterface) : boolean
    {
        let check : any = !!0;
        if(ticket.labels.length){
            for ( let lab in ticket.labels ){
                if(label.label_id == ticket.labels[lab]['label_id']){
                    return true
                }

            }
        }
        return check;
    }


    public changeCompanyTicketStatus( mailboxes : any,selectedTickets : any, currentStatus : string, oldStatus : string)
    {
        this.socket.emit('changeStatus',{
            selectedTickets:selectedTickets,
            room:this.subdomain,
            mailboxes:mailboxes,
            currentStatus:currentStatus,
            oldStatus:oldStatus});
    }

    public addSnooze(selectedTickets : any, snoozeData:any)
    {
        this.socket.emit('addSnooze',{
            selectedTickets:selectedTickets,
            room:this.subdomain,
            snoozeData:snoozeData});
    }



    public removeLabel( data : any  )
    {

        this.socket.emit('removeLabel',{
            data:data,
            room:this.subdomain
        });
    }

    public changeTicket(ticketId : number,user_id : number, ticket : TicketDtoInterface) : void
    {
        console.log('clientChangeTicket',this.ticketService.filterticketsStatus)
        this.socket.emit('clientChangeTicket',
            {
                room:this.subdomain,
                ticketId:ticketId,
                user_id  : user_id,
                ticket:ticket,
                oldStatus:this.ticketService.filterticketsStatus,
                mailboxes:this.mailboxService.mailboxes
            });
    }

    public sentNewLabels(label:LabelDtoInterface) : void{
        this.socket.emit('sentNewLabels',
            {
                room:this.subdomain,
                label:label
            });
    }

    public mergeLabel(ticketIds : Array<number>, label:LabelDtoInterface) : void
    {
        this.socket.emit('sentMergeLabels',
            {
                room:this.subdomain,
                ticketIds:ticketIds,
                label:label
            });
    }


    public changeInbox(count:number, status :any, mailboxId:number) : void
    {
        this.mailboxService.mailboxes.forEach( ( mailbox: DtoMailboxModel ,index : number) : void =>
        {
            if(mailbox.id == mailboxId && this.mailboxService.mailboxes[index][status+'_tickets_count'] + count >= 0){
                this.mailboxService.mailboxes[index][status+'_tickets_count'] = this.mailboxService.mailboxes[index][status+'_tickets_count'] + count;
            }
        });
    }

    public logoutUser() : void
    {
        this.socket.emit('logoutUser',
            {
                room: this.subdomain,
                userId: this.userModel.getLoggedUserId(),
                authToken: this.userModel.getToken()
            });
    }
}
