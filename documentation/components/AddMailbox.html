    public set settingFormErrors( currentErrors: any )
    {
        if ( _.empty(currentErrors) )
        {
            for ( let key in currentErrors )
            {
                if ( currentErrors.hasOwnProperty(key) && this.mailboxFormErrors.hasOwnProperty(key) )
                {
                    this.mailboxFormErrors[key].verify = !!1;
                    this.mailboxFormErrors[key].text   = currentErrors[key][0];
                }
            }
        }
    }

    @Input()
    public classSuccess: any = ( param: string ) : boolean => {
        return this.mailboxForm.controls[param].valid && this.mailboxForm.controls[param].touched;
    };

    @Input()
    public classError: any = ( param: string ) : boolean =>
    {
        if ( (! this.mailboxForm.controls[param].valid && this.mailboxForm.controls[param].touched && this.mailboxForm.submitted) ||
            (! this.mailboxForm.controls[param].valid && this.mailboxForm.submitted) )
        {
            return !!1;
        }
        else if ( this.mailboxFormErrors[param].verify && !this.mailboxForm.controls[param].valid  && this.mailboxForm.submitted)
        {
            return !!1;
        }
        else
        {
            return !!0;
        }
    };

    constructor
    (
        private formBuilder: FormBuilder,
        private ticketService: TicketsService,
        private router: Router,
        private mailboxService: MailBoxService,
        private mailboxModel: MailBoxModel,
        private toastr: ToastrService
    )
    {
        this.textEditor = new TextEditor()
    }

    public ngOnInit()
    {
        this.getAllAgents();
        
        this.mailboxForm = this.formBuilder.group(this.mailboxValidation);
        this.changeValues();
        this.textEditor.initEditorConfigs();
    }
    
    public changeValues(): void {
        this.mailboxForm['name'] = this.mailboxData.name;
        this.mailboxForm['email'] = this.mailboxData.email;
        this.mailboxForm['auto_reply'] = this.mailboxData.auto_reply;
        this.textEditorBody = this.mailboxData.signature;
    }

    public showSuccess() {

        this.toastr.success('Mailbox has been successfully created.', 'Success', this.defaultToastrParams);
    }

    public changeOpenedDropdwonLists( listName: string ) : void
    {
        let formdata = this.mailboxForm.value;
        Object.keys(this.dropdownOptions).map( ( index: string ) : void => {

            if ( index == listName ) {
                if (listName == 'autoReplyContentOn') {
                    this.dropdownOptions[index] = !!1;
                }
                if (listName == 'autoReplyContentOff') {
                    this.dropdownOptions.autoReplyContentOn = !!0
                }
                if (listName == 'bccContentOn') {
                    this.dropdownOptions[index] = !!1;
                }
                if (listName == 'bccContentOff') {
                    this.dropdownOptions.bccContentOn = !!0
                }
            }
        });
        this.mailboxForm.submitted = !!0;
    }
    
    

   public buildUpdatedData() : void
    {
        let newData: any;
            
        newData = <Object>(this.mailboxForm.value);
        if( this.dropdownOptions.autoReplyContentOn )
        {
            this.autoReply.on = "1";
            newData.auto_reply = this.autoReply.on;
        }
        else
        {
            this.autoReply.off = "0";
            newData.auto_reply = this.autoReply.off;
        }

        this.textEditorBody = this.textEditor.core.html.get();

        if( this.textEditorBody )
            newData.signature = this.textEditorBody;


        if( this.dropdownOptions.bccContentOn )
        {
            this.Bcc.autoBccOn = "1";
            newData.auto_bcc = this.Bcc.autoBccOn;
        }
        else
        {
            this.Bcc.autoBccOff = "0";
            newData.auto_bcc = this.Bcc.autoBccOff;
        }

        if(this.allowedUsers.length > 0)
            newData.allowed_users = this.allowedUsers;

    }
    public changeBuildUpdatedData() : void
    {
        let newData: any;

        newData = <Object>(this.mailboxForm.value);

        this.textEditorBody = this.textEditor.core.html.get();

        if( this.textEditorBody )
            newData.signature = this.textEditorBody;


        if( this.dropdownOptions.bccContentOn )
        {
            this.Bcc.autoBccOn = "1";
            newData.auto_bcc = this.Bcc.autoBccOn;
        }
        else
        {
            this.Bcc.autoBccOff = "0";
            newData.auto_bcc = this.Bcc.autoBccOff;
        }

        if(this.allowedUsers.length > 0)
            newData.allowed_users = this.allowedUsers;

    }
    public getAllAgents(): void
    {
        this.ticketService.getAgents()
            .then( ( data: any ) : void => {
                if ( data.success )
                {
                    this.allAgents = data.data;
                    if(this.allAgents.length > 0)
                    {
                        this.persons.forEach((person: any): void => {
                            person.verify = !!1;
                        });
                    }
                }
            });
    }

    public openAgents(personValue: any): void
    {
        if( personValue == 1)
            this.checkedAgents = !!1;
        else
            this.checkedAgents =!!0;
    }

    public aelectedAgents(checked: boolean, agent: any ,index:any): void
    {
        if (checked)
            this.allowedUsers.push(agent.id);
        else
            this.allowedUsers.splice(index, 1);
    }

    @Input()
    public onSubmitted() : void
    {
        this.mailboxForm.submitted = !!1;
        if(this.otherComponent){
            this.changeMailbox();
        }else{
            this.addMailbox();
        }
    }
    public addMailbox() : void
    {
        this.buildUpdatedData();
        this.mailboxForm.submitted = !!1;
        this.oneclick = !!1;
        this.mailboxService.createNewMailbox(this.mailboxForm.value)
            .then( ( data: any ) : void => {
                this.oneclick = !!0;
                if ( ! data.success )
                {
                    if ( ! data.errors.domain )
                        this.settingFormErrors = data.errors;
                }
                else
                {
                    this.mailboxForm.submitted = !!0;
                    let mailboxes = this.mailboxService.mailboxes;
                    mailboxes.push(data.data);
                    this.mailboxService.mailboxes.set(mailboxes);
                    this.mailboxModel.insert(data.data);
                    this.showSuccess();
                    this.dataToModal.dnsName = data.data.dns_name;
                    this.dataToModal.dnsValue = data.data.dns_value;
                    this.dataToModal.forwardAddress = data.data.forward_address;
                    this.dataToModal.mailboxId = data.data.id;
                    this.router.navigate(['/settings/mailbox']);
                    this.mailboxForm.submitted = !!0;
                }
            });
    }

    public changeMailbox(): void
    {
        this.oneclick = !!1;
        this.changeBuildUpdatedData();

        this.mailboxService.changeMailbox(this.mailboxForm.value, this.mailboxData.id)
            .then( ( data: any ) : void => {
                this.oneclick = !!0;
                if ( data.success )
                {

                    this.toastr.success('Mailbox has been successfully updated.', 'Success', this.defaultToastrParams);
                    this.mailboxForm.submitted = !!0;
                }
            });
    }

    public verifyModal()
    {
        this.mailboxService.verifyMailbox(this.dataToModal.mailboxId)
            .then( ( data: any ) : void => {

                if (data.success)
                {
                    this.router.navigate(['/mailbox']);
                    this.verifyModalOpening = !!0;
                }
                else
                {
                    this.verifyModalOpening = !!0;
                    this.router.navigate(['/mailbox']);
                }
           });
    }

    public insertVariables( $event : any , variable : any )
    {
        if (variable == 'signature')
        {
            this.textEditor.core.selection.restore();
            this.textEditor.core.html.insert($event.target.value, false);
            this.mailboxForm[variable] = this.textEditor.core.html.get();
        }
        else
            this.mailboxForm[variable] = (<any> document.getElementById(variable)).value + ' ' +  $event.target.value;
    }

    public closeVerifyModal()
    {
        this.dataToModal.dnsName        = '';
        this.dataToModal.dnsValue       = '';
        this.dataToModal.forwardAddress = '';
        this.dataToModal.mailboxId      = '';
        this.verifyModalOpening         = !!0;
    }
}
