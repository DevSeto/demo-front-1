import {Component, Input, OnInit, EventEmitter, Output, HostListener} from '@angular/core';
import {Title} from '@angular/platform-browser';
import {CannedrepliesService} from '../../../services/components/cannedreplies.service';
import {UsersModel} from '../../../models/components/users.model';
import {TextEditor} from '../../editor/editor';
import {Categorys} from '../../dto/canned-raply/dto/categorys.dto';
import {CannedReplies} from '../../dto/canned-raply/dto/canned-replies.dto';
import {CannedRepliesModel} from '../../dto/canned-raply/models/canned-replies.model'
import {Ticket} from "../../dto/ticket/dto/ticket.dto";
import {MailBoxModel} from '../../../models/components/mailbox.model';

declare let $ :any;

@Component({
    selector: 'create-canned-replies',
    templateUrl: '../html/create-canned-replies.html',
    providers: [CannedRepliesModel]
})

export class CreateCannedReplies implements OnInit {

    public textEditor:any;
    public cloud: any = showed;

    public selectionValue: string = 'Select category';
    public categoryName: string;
    public previousTitle: string;
    public modalName = 'Create new';
    public createCategory: boolean = !!0;

    public newCategoryData : any  = {
        newCategory:'',
        newCategoryError:!!0
    };

    public cannedRepliesFormValidation : any = {
        category_id : !!0,
        body:!!0
    };

    public selectMenu: any = {
        selectByTime: <boolean>!!0,
        showCannedRepliesList: <boolean>!!0,
        selectCategory: <boolean>!!0
    };
    public textEditorBody: any;

    constructor
    (
        private title: Title,
        private usersModel:UsersModel,
        public cannedRepliesService:CannedrepliesService,
        public cannedRepliesModel: CannedRepliesModel,
        private ticket: Ticket,
        private mailboxModel: MailBoxModel
    )
    {
        this.textEditor = new TextEditor(usersModel.getToken());
    }

    public ngOnInit()
    {
        if(this.cannedRepliesService.cannedrepliesBody){
            this.modalName = 'Save';
        }
        this.cannedRepliesModel.body = this.cannedRepliesService.cannedrepliesBody;
        this.previousTitle = this.title.getTitle();
        this.textEditor.initEditorConfigs();
        this.getAllCategorys();

    }


    @HostListener('document:keyup', ['$event'])
    public isPressedToEscape( event: any ) : void
    {
        if ( event.keyCode == 27 )
            this.cloud.effect = 'createCannedReplies';
    }

    public getAllCategorys(): void {
        this.cannedRepliesService.getAllCategory()
            .then((category : ResponseData<Categorys> ) =>
            {
                if(category.success)
                    this.cannedRepliesService.categorys.set(category.data);
            })
    }

    public createCannedReplies() : void
    {
        this.cannedRepliesModel.body = this.textEditor.core.html.get();
        this.cannedRepliesModel.mailbox_id = this.mailboxModel.getMailboxCurrentId()

        if( this.submitValidation()){
            this.cloud.effect = 'repliesSpin';
            this.cannedRepliesService.createCannedReplies(this.cannedRepliesModel)
                .then((data : any) =>
                {
                    if(data.success)
                    {
                        data.data.category = {};
                        data.data.category.name = this.categoryName;
                        this.cannedRepliesService.cannedReplies.unshift(data.data);
                        this.cloud.effect = 'createCannedReplies'
                    }
                    this.cloud.effect = 'repliesSpin';
                })
        }
    }

    public submitValidation(): boolean
    {
        let verify: boolean = !!1;

        Object.keys(this.cannedRepliesModel).forEach( ( indexName: string ) : void => {

            if ( indexName == 'body' )
            {
                if(this.cannedRepliesModel.body.length  < 0)
                {
                    this.cannedRepliesFormValidation[indexName] = 'Field is required.';
                    verify = !!0;
                }
                else
                    this.cannedRepliesFormValidation[indexName] = null;
            }

            if ( indexName == 'name' )
            {
                if(this.cannedRepliesModel.name.length < 1){
                    this.cannedRepliesFormValidation[indexName] = 'Field is required.';
                    verify = !!0;
                }
                else
                    this.cannedRepliesFormValidation[indexName] = null;
            }
            if ( indexName == 'category_id' )
            {
                if(!this.cannedRepliesModel.category_id){
                    this.cannedRepliesFormValidation[indexName] = 'Field is required.';
                    verify = !!0;
                }
                else
                    this.cannedRepliesFormValidation[indexName] = null;
            }
        });
        return verify;
    }

     /***
     * replies Categors
     */

    public createCategoyModal( status : boolean) : void
    {
        this.createCategory = status;
    }

    public addCategory() : void
    {
        if( this.newCategoryData.newCategory.length )
        {
            this.newCategoryData.newCategoryError = !!0;
            this.cannedRepliesService.setNewCategory(this.newCategoryData.newCategory)
                .then((data : any) => {
                    if(data.success)
                    {
                        this.newCategoryData.newCategory = '';
                        this.createCategory = !!0;
                        this.cannedRepliesService.categorys.push(data.data);
                        this.selectCategory(data.data.id, data.data.name)
                    }
                })
        }
        else
        {
            this.newCategoryData.newCategoryError = !!1;
        }
    }

    public cusstomErrorValidation(value : any) : void
    {
        if(value.length > 0)
            this.newCategoryData.newCategoryError = !!0;
        else
            this.newCategoryData.newCategoryError = !!1;
    }

    public cusstomValidationFromReplies(value : any , name : any) : void
    {
        if(value.length > 0)
            this.cannedRepliesFormValidation[name] = !!0;
        else
            this.cannedRepliesFormValidation[name] = !!1;

    }

    public setCannedReplies( index : any ) : void
    {
        this.textEditor.core.selection.restore();
        this.textEditor.core.html.insert(this.cannedRepliesService.cannedReplies[index].body, false);
        this.openCloseMenu(['showCannedRepliesList'], !!1);
    }

    public openCloseMenu( menuNames: Array<string>, clickOutside?: boolean, callFunction?: string ) : void
    {
        menuNames.forEach( ( name: string ) : void => {
    
            if( callFunction )
            {
                this[callFunction](name, clickOutside);
            }
            else if ( clickOutside )
            {
                if ( this.selectMenu[name] )
                    this.selectMenu[name] = !!0;
            }
            else
            {
                this.selectMenu[name] = !this.selectMenu[name];
            }
        })
    }

    public selectCategory( id: number, name: string ) : void
    {
        this.cannedRepliesModel.category_id = id;
        this.selectionValue = name;
        this.categoryName = name;
        this.cannedRepliesFormValidation.category_id = !!0;
    }

    public createNewCategory() : void
    {
        this.createCategoyModal(!!1);
    }

}
