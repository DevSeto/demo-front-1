    constructor(
        private userService: UsersService,
        private notificationsService: NotificationsService,
        private toastr: ToastrService
    )
    {

    }


    public ngOnInit() : void
    {
        this.notificationsService.getNotificationSetings()
            .then((data:any)=>{
                if(data.success){
                    (<any>Object).values(data.data).forEach((nots: any, index:any ): void => {
                        if(this.notificationFormData[index] && this.notificationFormData[index]['condition_id'] == nots.condition_id){
                            this.notificationFormData[index]['email'] = !!nots.email;
                            this.notificationFormData[index]['browser'] = !!nots.browser;
                        }
                        
                        if(!(!this.notificationSetingData['conditions'][this.notificationFormData[index]['condition_id']]  && this.notificationSetingData['conditions'][this.notificationFormData[index]['condition_id']] != 0 ) ){
                            this.notificationSetingData['conditions'][this.notificationFormData[index]['condition_id']]['email'] = nots.email;
                            this.notificationSetingData['conditions'][this.notificationFormData[index]['condition_id']]['browser'] = nots.browser;
                        }
                    });
                }
        })
    }


    public clickCheckbox( event : any, id : any, emailOrBrowser: string) : void
    {
        if(event.target.checked){
            this.notificationSetingData['conditions'][id][emailOrBrowser] = 1;
        }else{
            this.notificationSetingData['conditions'][id][emailOrBrowser] = 0;
        }
    }
    
    public changeNotificationSetings() : void
    {
        this.notificationsService.openNotificationAccesses(this.notificationSetingData)
            .then((data:any)=>{
                if (data.success)
                    this.toastr.success('Notification settings successfully saved.', 'Success', <any>{ progressBar: true, positionClass: "toast-bottom-right"});
            })
    }




    public enableNotification()
    {
        Notification.requestPermission().then((result) =>
        {
            if(result == 'denied'){
                alert('please enable your notification from your browser settings!!axmax');
            }else if(result == 'granted'){
                this.notificationsService.getPushNotificationData();
            }
            console.log(result);
        })

    }

    public notificationIsEnable(): boolean
    {
        if ((<any>Notification).permission !== 'granted')
        {
            return false;
        }

        return true;

    }
}
