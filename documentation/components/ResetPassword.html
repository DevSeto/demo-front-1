import {Component, OnInit, Output, EventEmitter, Input} from "@angular/core";
import {UsersService} from "../../services/components/users.service";
import {Router}    from '@angular/router';
import {Title} from '@angular/platform-browser';
import {ToastrService} from '../../modules/toastr';


@Component({
    selector: 'reset-password',
    templateUrl: '../../html/reset-password/reset-password.html',
})
export class ResetPassword implements OnInit {

    private defaultToastrParams: any = <any> {
        progressBar: true,
        positionClass: "toast-bottom-right",
    };


    @Input('individualUser')
    public individualUser:any;

    public previousTitle:string;
    public resetForm:any;

    public resetData:any = {
        old_password: "",
        password: "",
        password_confirmation: ""
    };
    public resetFormValidation:any = {
        old_password: '',
        password: '',
        password_confirmation: ''
    };

    @Output('closeResetEmit')
    public closeResetEmit:EventEmitter<void|boolean> = new EventEmitter<void|boolean>();

    constructor(private usersService:UsersService,
                private router:Router,
                private titleService:Title,
                private toastr: ToastrService
    )
    {
    }

    public ngOnInit() {
    }

    public resetValidatia():boolean {
        let verify:boolean = !!1;
        Object.keys(this.resetData).forEach((index:string):void => {
            if (index == 'old_password') {
                if (this.resetData.old_password.length < 6 && !this.resetData.old_password.match(/^(?=.*[0-9])(?=.*[a-zA-Z])([a-zA-Z0-9]+)$/)) {
                    this.resetFormValidation.old_password = 'Invalid curront password. Password must be at least 6 characters long and contain a number.';
                    verify = !!0;
                    console.log('old_password');
                    return;
                }
                else {
                    this.resetFormValidation.old_password = null;
                }

            }
            if (index == 'password') {
                if (this.resetData.password.length < 6 && !this.resetData.password.match(/^(?=.*[0-9])(?=.*[a-zA-Z])([a-zA-Z0-9]+)$/)) {
                    this.resetFormValidation.password = 'Invalid password. Password must be at least 6 characters long and contain a number.';
                    verify = !!0;
                    console.log('password');
                    return;
                }
                else {
                    this.resetFormValidation.password = null;
                }

            }
            if (index == 'password_confirmation') {
                if (this.resetData.password_confirmation.length < 6 && !this.resetData.password_confirmation.match(/^(?=.*[0-9])(?=.*[a-zA-Z])([a-zA-Z0-9]+)$/)) {
                    this.resetFormValidation.password_confirmation = 'Invalid confirm password. Password must be at least 6 characters long and contain a number.';
                    verify = !!0;
                    console.log('password_confirmation');
                    return;
                }
                else {
                    this.resetFormValidation.password_confirmation = null;
                }

            }
        });
        return verify;
    }

    public valueIsPassword( name: any): void
    {
        if (this.resetData[name].length == 0)
        {
            this.resetFormValidation[name] = 'Field is required.';
        }

        if (name == 'password_confirmation')
        {
            if(this.resetData.password_confirmation !== this.resetData.password)
            {
                this.resetFormValidation.password_confirmation = "The passwords don't match."
            }
        }
    }

    public valueIsntPassword(name: any): void
    {
        this.resetFormValidation[name] = null;
    }

    public closeResetPopup(): void {
        this.closeResetEmit.emit();
        this.titleService.setTitle(this.previousTitle);
    }

    public showSuccess() {
        this.toastr.success('Reset password email successfully send', 'Success', this.defaultToastrParams);
    }

    public passwordReset(): void
    {
        this.resetData.id = this.individualUser.id;

        if ( this.resetValidatia() )
        {
            this.usersService.resetUserPassword(this.resetData)
                .then( ( data: any ) => {
                    if ( data.success)
                    {
                        console.log(data.data);
                        this.showSuccess();
                        this.closeResetEmit.emit();
                    }
                })
        }
    }
}
