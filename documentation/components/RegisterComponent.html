import {Component, Input, OnInit, ViewChild} from "@angular/core";
import {Router} from '@angular/router';
import {Title} from '@angular/platform-browser';
import {FormBuilder, Validators } from '@angular/forms';
import {UsersService} from '../../services/components/users.service';
import {GlobalServices} from "../../services/helpers/global.service";
import {ValidationService} from "../../services/helpers/validation.service";
import {UsersModel} from "../../models/components/users.model";
import {ReCaptchaComponent} from '../captcha/captcha.component';
import {_} from "../../services/helpers/helper.service";
import {MailBoxModel} from '../../models/components/mailbox.model';
import {User} from '../../dto';

@Component({
	templateUrl: '../../html/register/register.component.html'
})

export class RegisterComponent implements OnInit {

    public buttonpadding: boolean = !!0;
    public inputType: string = 'password';
    public showHide: boolean = !!0;
    public showHideText: string = 'Show';

    @ViewChild(ReCaptchaComponent)
    public captcha: ReCaptchaComponent;

    private readonly _googleRecaptchaKey: string = '6LfEiU0UAAAAANDN_zdOdapm8Jh9p8pOaBwThJt_';
    private _googleRecaptchaResolved: boolean = !!0;

    public formSubmit: boolean = !!0;
    public oneclick: boolean = !!0;

    public passwordInclude: Array<any> = [
        {
            name: 'One lowercase character',
            active: !!0
        },
        {
            name: 'One uppercase character',
            active: !!0
        },
        {
            name: 'One number',
            active: !!0
        },
        {
            name: 'One special character',
            active: !!0
        },
        {
            name: '6 characters minimum',
            active: !!0
        }
    ];

    public defoultTrue: boolean = !!0;
    private _userFormErrors: any = {
        full_name: {
            verify: !!0,
            text: _._
        },
        password: {
            verify: !!0,
            text: _._
        },
        email: {
            verify: !!0,
            text: _._
        },
        company_url: {
            verify: !!0,
            text: _._
        }
    };

    private userValidation: any = {
        full_name: [
            _._,
            [
                Validators.required,
                ValidationService.fullNameValidator,
                Validators.minLength(3),
            ]
        ],
        password: [
            _._,
            [
                Validators.required,
                ValidationService.passwordValidator,
                Validators.minLength(6)

        ],
        email: [
            _._,
            [

                ValidationService.emailValidator,
                Validators.minLength(6)

        company_url: [
            _._,

                Validators.minLength(3)

        ]
    };

    public userForm: any;

    public get googleRecaptchaKey() : string
    {
        return this._googleRecaptchaKey;
    }

    private set googleRecaptchaResolved( verify: boolean )
    {
        this._googleRecaptchaResolved = verify;
    }

    private get googleRecaptchaResolved() : boolean
    {
        return this._googleRecaptchaResolved;
    }

    public get userFormErrors() : any
    {
        return this._userFormErrors;
    }

    public set userFormErrors( currentErrors: any )
    {
        if ( _.empty(currentErrors) )
        {
            for ( let key in currentErrors )
            {
                if ( currentErrors.hasOwnProperty(key) && this._userFormErrors.hasOwnProperty(key) )
                {
                    this._userFormErrors[key].verify = !!1;
                    if(typeof currentErrors[key] == 'string'){
                        this._userFormErrors[key].text = currentErrors[key];

                    }else{
                        this._userFormErrors[key].text = currentErrors[key][0];

                    }
                }
            }
        }
    }

    @Input()
    public register: any = () : void => {

        if ( this.userForm.dirty && this.userForm.valid )
        {
            if ( this.userForm.submitted )
                delete this.userForm.submitted;

            // console.log(this.oneclick ,'this.oneclick ')
            // if ( ! this.oneclick )
            // {

                this.verifyGoogleRecaptcha();
            // }
        }
        else
        {
            this.userForm.submitted = !!1;
        }
    };

    @Input()
    public classError: any = ( param: string ) : boolean => {
        if ( (! this.userForm.controls[param].valid && this.userForm.controls[param].touched && this.userForm.submitted) ||
            (! this.userForm.controls[param].valid && this.userForm.submitted) )
        {
            this.buttonpadding = !!1;
            return !!1;
        }
        else if ( this.userFormErrors[param].verify  && this.userForm.submitted)
        {
            return !!1;
        }
        else
        {
            this.buttonpadding = !!0;
            return !!0;
        }
    };

    @Input()
    public classSuccess: any = ( param: string ) : boolean => {
        return this.userForm.controls[param].valid && this.userForm.controls[param].touched;
    };

    @Input()
    public verifyTwinsError( param: string, value: string )
    {
        if ( _.empty(this.userFormErrors[param].text) )
        {
            if ( this.userFormErrors[param].text == value )
                this.userFormErrors[param].verify = !!1;
            else
                this.userFormErrors[param].verify = !!0;
        }
    }

    constructor (
        private router: Router,
        private usersService: UsersService,
        private titleService: Title,
        private mailboxModel: MailBoxModel,
        private formBuilder: FormBuilder,
        private userModel: UsersModel,
        private user:User
    ) { }

    public ngOnInit() : void
    {

        this.titleService.setTitle('Birddesk: Register');
        this.userForm = this.formBuilder.group(this.userValidation);
    }

    private createUser() : void
    {
        // console.log('asdasdasdsd');
        this.splitFullName();
        let locationPaths: Array<string> = (<any>window).location.host.split('.');
        this.userModel.userAuthotization = !!0;
        this.usersService.registerUser(this.userForm.value)
            .then( ( data: any ) => {
                if ( ! data.success )
                {
                    this.oneclick = !!0;
                    this.userFormErrors = data.errors;
                }
                else
                {
                    this.userModel.setCookiAfterReg();
                    if ( locationPaths[0] == data.data.company_url )
                    {
                        this.userModel.userAuthotization = !!1;

                        this.mailboxModel.insertMailboxId(data.data.mailbox_id || 1);
                        this.userModel.insert(data.data);
                        this.user.step = !!1;
                        this.defoultTrue = !!1;
                        this.router.navigate(['/ticket']);
                        // this.router.navigateByData({
                        //     url: ["/ticket"],
                        //     data: [this.defoultTrue]
                        // });
                    }
                    else
                    {
                        (<any>window).location.href = GlobalServices.getCompanyUrl(data.data.company_url);
                    }
                }
            }).catch(() : void =>{
                this.oneclick = !!0;
            });
    }

    public splitFullName() : void
    {
        let userFullName: Array<string>;

        if ( this.userForm.value.full_name )
        {
            userFullName = this.userForm.value.full_name.split(' ');

            userFullName.forEach( ( value: string, index: number ) : void => {

                if ( ! index )
                {
                    this.userForm.value.first_name = value;
                }
                else
                {
                    if ( ! this.userForm.value.last_name )
                        this.userForm.value.last_name = '';

                    this.userForm.value.last_name += ' ' + value;
                }
            });
        }

        delete this.userForm.value.full_name;
    }

    public verifyGoogleRecaptcha( token?: string ) : void
    {
        if ( ! this.googleRecaptchaResolved )
        {
            if ( _.empty(token) || _.empty(this.captcha.getResponse()) )
            {
                this.googleRecaptchaResolved = !!1;
                this.oneclick = !!1;

                this.createUser();
            }
            else
            {
                this.captcha.execute();
            }
        }
        else
        {
            this.googleRecaptchaResolved = !!0;
            this.captcha.reset();

            return this.verifyGoogleRecaptcha();
        }
    }

    public toggleShow() : void
    {
        this.showHide = !this.showHide;
        if (this.showHide){
            this.inputType = "text";
            this.showHideText = 'Hide';
        }
        else {
            this.inputType = "password";
            this.showHideText = 'Show';
        }
    }

    public passwordIncludes( password: any) : void
    {
        let validCount = 0,
            strengthPasswordText = document.getElementById('strength-text'),
            strengthProgressWidth = document.getElementById('progress-width');

        if ( password.match(/^(?=.*[a-z])/) )
            this.passwordInclude[0].active = !!1;
        else
            this.passwordInclude[0].active = !!0;

        if ( password.match(/^(?=.*[A-Z])/) )
            this.passwordInclude[1].active = !!1;
        else
            this.passwordInclude[1].active = !!0;

        if ( password.match(/^(?=.*[0-9])/) )
            this.passwordInclude[2].active = !!1;
        else
            this.passwordInclude[2].active = !!0;

        if ( password.match(/^(?=.*[!@#\$%\^&\*])/) )
            this.passwordInclude[3].active = !!1;
        else
            this.passwordInclude[3].active = !!0;

        if ( password.match(/^(?=.{6,100})/) )
            this.passwordInclude[4].active = !!1;
        else
            this.passwordInclude[4].active = !!0;

        this.passwordInclude.forEach( (value: any) => {
            if (value.active)
                validCount++;
        });

        if (password.length)
        {
            switch(validCount){
                case 5:
                    strengthPasswordText.innerText = "Strong";
                    strengthPasswordText.style.color = "green";
                    strengthProgressWidth.style.width = "100%";
                    strengthProgressWidth.style.backgroundColor = "green";
                    break;
                case 4:
                case 3:
                    strengthPasswordText.innerText = "So-so";
                    strengthPasswordText.style.color = "orange";
                    strengthProgressWidth.style.width = "60%";
                    strengthProgressWidth.style.backgroundColor = "orange";
                    break;
                default:
                    strengthPasswordText.innerText = "Weak";
                    strengthPasswordText.style.color = "red";
                    strengthProgressWidth.style.width = "30%";
                    strengthProgressWidth.style.backgroundColor = "red";
                    break;
            }
        }
    }
}
