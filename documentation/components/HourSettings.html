import {Component, OnInit, Output, EventEmitter} from '@angular/core';
import {FormBuilder} from '@angular/forms';
import {MailBoxService} from '../../services/components/mailbox.service';
import {ActivatedRoute, Router} from '@angular/router';
import {ToastrService} from '../../modules/toastr/toastr/toastr-service';

declare let $: any;

@Component({
    selector: 'hour-settings',
    templateUrl: '../../html/hour-settings/hour-settings.html'
})

export class HourSettings implements OnInit {

    public isShowed: boolean = !!0;
    public mailboxId: number;
    public selectMenu: any = {};
    public cloud: any = showed;
    private defaultToastrParams: any = <any> {
        progressBar: true,
        positionClass: "toast-bottom-right",
    };
    public timeDatattt: any = {

        from: new Date(0, 0, 0, 9, 0, 0),

    };
    public timeData: any = {
        'Monday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        },
        'Tuesday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        },
        'Wednesday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        },
        'Thursday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        },
        'Friday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        },
        'Saturday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        },
        'Sunday': {
            from: new Date(0, 0, 0, 9, 0, 0),
            to: new Date(0, 0, 0, 17, 0, 0),
            enable: 0
        }
    };

    public days:Array<Object> = [
        'Monday',
        'Tuesday',
        'Wednesday',
        'Thursday',
        'Friday',
        'Saturday',
        'Sunday',
    ];

    public momentValue:any;
    //
    public date: Date = new Date('Jul 16, 2018, 6:15 PM');
    public settings = {
        startDate: new Date(2016, 5, 10),
        autoclose: true,
        todayBtn: 'linked',
        todayHighlight: true,
        assumeNearbyYear: true,
        format: 'D, d MM yyyy'
    }

    @Output('closeHoursEmit')
    public closeHoursEmit:EventEmitter<void|boolean> = new EventEmitter<void|boolean>();

    constructor(
        private route: ActivatedRoute,
        private router: Router,
        private formBuilder: FormBuilder,
        private toastrService: ToastrService,
        private mailboxService: MailBoxService
    )
    {
        this.momentValue = new Date();
        const routeParams: any = this.route.snapshot.params;
        if ( routeParams['id'] )
        {
            this.mailboxId = routeParams['id'];
        }
        this.days.forEach( (from: any, idx: number) => {
            this.selectMenu[`from${idx}`] = !!0;
            this.selectMenu[`to${idx}`] = !!0;
        });
    }

    public ngOnInit()
    {
        this.mailboxService.getOfficeHours( this.mailboxId )
            .then( (data: any) => {
                if ( data.success ) {
                    (<any>Object).keys(data.data).forEach((day: any, index: string) => {
                        this.timeData[day]['to'] = (new Date(data.data[day]['to']));
                        this.timeData[day]['from'] = (new Date(data.data[day]['from']));
                        this.timeData[day]['enable'] = data.data[day]['enable'];
                    })
                }
                this.isShowed = !!1;

                    // this.timeData = data.data;
            });
    }

    public updateHours() : void
    {
        let validTimeCount: number = 0;

        (<any>Object).keys(this.timeData).forEach( (day: any, index : string) =>
        {
            this.timeData[day]['to'] = (new Date(this.timeData[day]['to']));
            this.timeData[day]['from'] = (new Date(this.timeData[day]['from']));
            //
            if ( this.timeData[day]['to'] <= this.timeData[day]['from'])
            {
                this.toastrService.info(
                    "You can't do this!",
                    'Info',
                    {
                        progressBar: true,
                        positionClass: "toast-bottom-right"
                    });
            }
            else
                validTimeCount += 1;
        });

        if ( validTimeCount == (<any>Object).values(this.timeData).length )
        {
            this.mailboxService.setOfficeHours( this.timeData, this.mailboxId )
                .then( (data: any) => {
                    if ( data.success )
                    {
                        this.showSuccess();
                        this.cloud.effect = 'hourSettings';
                    }
                    else
                    {
                        if ( data.errors.timezone )
                            this.router.navigate(['/settings/company-profile']);
                    }
                });
        }
    }

    public showSuccess()
    {
        this.toastrService.success('Hours settings successfully updated.', 'Success', this.defaultToastrParams);
    }
    public clickCheckbox( day: string ) : void
    {
        this.timeData[day].enable = (-(this.timeData[day].enable-1)) << 0;
    }

    public closeHoursPopup() : void
    {
        this.cloud.effect = 'hourSettings';
    }

    public openCloseMenu( menuNames: Array<string>, clickOutside?: boolean, callFunction?: string ) : void
    {
        menuNames.forEach( ( name: string ) : void => {

            if( callFunction )
            {
                this[callFunction](name, clickOutside);
            }
            else if ( clickOutside )
            {
                if ( this.selectMenu[name] )
                {
                    this.selectMenu[name] = !!0;
                }
            }
            else
            {
                this.selectMenu[name] = !this.selectMenu[name];
            }
        })
    }
}
