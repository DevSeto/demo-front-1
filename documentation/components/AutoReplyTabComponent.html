    {
        const routeParams: any = this.route.snapshot.params;
        this.previousRoute = GS.lastRoute.last;

            this.mailboxId = routeParams['id'];
            this.previousTitle = `Back <- ${this.title.getTitle()}`;



            return !!1;

        }
        return !!0;
    }

    public buildUpdatedData() : void
    {
        let newData: any;

        newData = <Object>(this.replyData);

        if ( !this.autoReplyOnOff ){

            this.replyData.auto_reply = "0";
            newData.auto_reply = this.replyData.auto_reply;
        }
        // console.log(this.autoReplyForm.controls.auto_reply_body.value);
        if(!this.replyData.auto_reply_body && !this.autoReplyForm.controls.auto_reply_body.value && this.autoReplyForm.submitted && this.auto_reply_body_valid && this.autoReplyOnOff){
            this.auto_reply_body_valid = !!0;
        }else{
            this.autoReplyForm.controls.auto_reply_body.setValue(this.textEditor.core.html.get());
        }
    }

    @Input()
    public saveChanges(): void
    {
        // console.log(this.replyData);
        //
        // console.log(this.autoReplyOnOff);
        // console.log(this.replyData.auto_reply);
        // console.log(this.textEditor.core.html.get())
        this.autoReplyForm.submitted = !!1;
        // this.replyValidation();
        //093814500 aro

        if ( this.autoReplyOnOff ) {
            this.replyData.auto_reply_body = this.textEditor.core.html.get()
            this.buildUpdatedData();
            if (this.autoReplyForm.valid) {

                this.mailboxService.changeAutoreply(this.replyData, this.mailboxId)
                    .then((data:any):void => {
                        if (data.success) {
                            this.showSuccess();
                            this.changeAutoReplyemit.emit(this.replyData);
                            this.autoReplyOnOffEmit.emit(this.autoReplyOnOff);
                            // this.router.navigateByData({
                            //     url: ["/ticket"],
                            //     data: this.mailboxId
                            // });
                        }
                    });
            }
        }
        else
        {
            this.mailboxService.changeAutoreply(this.replyDataOff, this.mailboxId)
                .then((data:any):void => {
                    if (data.success) {
                        this.showSuccess();

                        this.replyData =  {
                            auto_reply: '',
                            auto_reply_subject: '',
                            auto_reply_body: ''
                        };
                        this.changeAutoReplyemit.emit(this.replyData);
                        this.autoReplyOnOffEmit.emit(this.autoReplyOnOff);
                        // this.router.navigateByData({
                        //     url: ["/ticket"],
                        //     data: this.mailboxId
                        // });
                    }
                });
        }
    }

    public getMailbox( ): void
    {
        this.mailboxService.getMailboxById(this.mailboxId)
            .then(( data: any ) : void => {

                if ( data.success )
                {

                    this.mailboxdata = data.data;
                    this.succesForForward = !0;
                    this.succesTrue = !0;
                    this.autoReplyData();
                }
            });
    }



    public insertVariables( $event : any , variable : any )
    {
        let text:any = this.replyData[variable];
        let p : any = '';
        if(text != ''){
            text = this.replyData[variable].replace('</p>', ' ');
            p = '</p>';
        }
        this.variable = text +   $event.target.value + p;
        this.replyData[variable] = this.variable;
    }

    public showSuccess()
    {
        this.toastr.success('successfully updated.', 'Success', this.defaultToastrParams);
    }

}
