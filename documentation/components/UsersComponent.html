    public subscriptions() : void
    {
        this.subscription = this.userservice.usersSortingSatatus.subscribe(( event: any ) : void => {
            this.userName  = event;
            this.sortedUser(event.status);
        });

        this.userservice.usersAfterInvite.subscribe( ( status: boolean ) : void => {
            if (status)
                this.getUsers()
        });
    }

    public sortedUser( active: string )
    {
        if ( active )
        {
            this.clicksorted = !!1;
            this.allUsers = [];
            this.getUsers();
        }
    }

    public getUsers(): void {
        
        let acount_one: any,
            confirmedUsers: any,
            invitedUsers: any;
        
        this.userservice.getAllUsers({active: 'active'})
            .then( ( data: any ) : void => {

                if (data.success)
                {
                    // if ( data.data.invitedUsers.length > 0 && this.firstInit)

                    if(this.userservice.userParams.active  == 'active')
                    {
                        acount_one = [data.data.confirmedUsers[0]];
                        confirmedUsers = data.data.confirmedUsers;
                        confirmedUsers.splice(0,1);
                        
                        this.children = confirmedUsers;
                        this.children = this.children.sort(function(a: any,b: any)
                        {
                            return new (<any>Date)((<any>b).created_at) - new (<any>Date)((<any>a).created_at);
                        });
                        this.children = acount_one.concat(this.children);
                    }
                    else if(this.userservice.userParams.active  == 'inactive' )
                    {
                        invitedUsers = data.data.invitedUsers;
                        (<any>Object).values(invitedUsers).forEach((obj: any, key: any): void =>
                        {
                            if(obj.role_id == 4)
                            {
                                invitedUsers[key].display_user_role = "Agent";
                            }
                            else if(obj.role_id == 3)
                            {
                                invitedUsers[key].display_user_role = "Admin";
                            }
                            invitedUsers[key].status = 1;
                        });
                        
                        this.children = invitedUsers.sort(function(a: any,b: any)
                        {
                            return new (<any>Date)((<any>b).created_at) - new (<any>Date)((<any>a).created_at);
                        });
                    }
                    this.allUsers = this.children;

                    this.usersFilter();
                    // this.rerenderDsersList();
                }
            });
    }

    public usersFilter(): void {

        if (typeof this.userName == 'string')
        {
            this.allUsers.forEach((user: any): void => {
                if(user.display_user_role == this.userName)
                {
                    this.allUsers = [];
                    this.userFiltered.push(user)
                }
            });
            this.allUsers = this.userFiltered;
            this.selectMenu.emptyFolderMsg = this.allUsers.length == 0 ? !!1 : !!0;
        }
        if ( this.clicksorted == !!1 )
        {
            this.allUsers = [];
            this.allUsers = this.children;
            this.selectMenu.emptyFolderMsg = this.allUsers.length == 0 ? !!1 : !!0;
        }
    }

    public redirectUserProfilePage( user: any ) : void
    {
        if(this.userservice.userParams.active == 'active' && user.status !== 1 )
        {
            this.router.navigate(['/settings/users-profile', user.id]);
        }
    }

    public openCloseMenu( menuNames: Array<string>, clickOutside?: boolean, callFunction?: string ) : void
       {
        // this.closeAddUser = !!1;

        menuNames.forEach( ( name: string ) : void => {

            if( callFunction )
            {
                this[callFunction](name, clickOutside);
            }
            else if ( clickOutside )
            {
                if ( this.selectMenu[name] )
                    this.selectMenu[name] = !!0;
            }
            else
                this.selectMenu[name] = !this.selectMenu[name];
        });
    }
