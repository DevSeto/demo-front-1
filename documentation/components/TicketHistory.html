    public getTicketHistory()
    {
        this.ticketsService.getTicketHistory(this.ticketsService.ticket.id)
            .then((historyData:any)=> {
                this.ticketsService.individualHistory.set(historyData.data);
                this.ticketsService.individualHistory.reverse();
                // this.ticketHistory = historyData.data;
                // this.ticketHistory.reverse();
            });
    }
