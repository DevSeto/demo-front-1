    public getAllNotification(): void
    {
        let offSet : number = (this.ticketService.ticketsNotification['notifications'])?this.ticketService.ticketsNotification['notifications'].length:0;

        if (this.mailboxId && this.inprogress && this.ticketService.ticketsNotification.total != offSet) {
            this.inprogress = !!0;
            this.notificationsService.getAllNotification(offSet)
                .then((data: any): void => {

                    if (data.success ) {
                        if(this.ticketService.ticketsNotification.notifications){
                            // data.data.notifications.notification_id = data.data.notifications.id
                            data.data.notifications = this.ticketService.ticketsNotification.notifications.concat(data.data.notifications);
                        }
                        this.ticketService.ticketsNotification.set(data.data);
                        console.log('this.ticketService.ticketsNotification',this.ticketService.ticketsNotification)
                        this.inprogress = !!1;
                    }
                })
        }

    }

    public scrollingNotification( event : any ) : void
    {
        if(event.target.scrollHeight - event.target.scrollTop == event.target.clientHeight && this.inprogress ){
            this.getAllNotification();
            // event.target.scrollTop = event.target.scrollTop - 100;
            // document.getElementsByClassName('list-unstyled history-list table-scroll addscroll-box')[0].scrollTop = document.getElementsByClassName('list-unstyled history-list table-scroll addscroll-box')[0].scrollTop - 100;
        }

    }

    public redirectToTicketPage( ticket : TicketDtoInterface, idx:number ) : void
    {

        console.log(ticket)
        if(ticket.is_viewed == 0 && this.ticketService.ticketsNotification.total_new > 0){
            if(!ticket.id){
                this.notificationsService.setViewedNotification(ticket.notification_id);
            }else{
                this.notificationsService.setViewedNotification(ticket.id);
            }
            this.ticketService.ticketsNotification.total_new = this.ticketService.ticketsNotification.total_new - 1;

            this.ticketService.ticketsNotification.notifications[idx]['is_viewed'] = 1;
        }
        this.router.navigate(['/ticket',ticket.notifications.ticket_id])

    }


    public redirectToMailbox():void
    {

        this.router.navigate(['/mailbox-settings/'+ this.mailBoxModel.getMailboxCurrentId() +'/2']);
    }
}
